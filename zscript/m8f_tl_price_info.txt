
class m8f_tl_PriceInfo : Inventory
{

  m8f_tl_PriceInfo init(string currency, int price)
  {
    _currency = currency;
    _price    = price;
    _marked   = 0;

    return self;
  }

  override void Tick()
  {
    if (owner == null) return;

    console.printf("tick: %s %d %d", owner.gettag(), owner.pos.x, owner.pos.y);
  }

  // private attribute section

  private string _currency;
  private int    _price;
  private int    _marked;

} // class m8f_tl_PriceInfo
